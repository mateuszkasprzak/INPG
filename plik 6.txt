
#include <stdlib.h>
#include <math.h>


int rownanie(float a,float b,float c,float *x1,float *x2)
{
    double delta=0;
    delta=(b*b-4*a*c);
    if(a==0)
    {
        return 1;
    }
    else if(delta>0)
    {
        *x1=(-b-sqrt(delta))/(2*a);
        *x2=(-b+sqrt(delta))/(2*a);
        return 2;
    }
    else if(delta==0)
    {
        *x1=-b/(2*a);
        return 3;
    }
    else
    {
        return 0;
    }
}

{   float a=0, b=0, c=0;
